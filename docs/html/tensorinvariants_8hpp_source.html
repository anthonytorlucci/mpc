123  // alpha = trace(tensor_ij)

130  T StressTensorContraction(const mpc::core::StressTensor<T>& sigma_ij) {
131  // sigma_kk = (sigma_ik) * (kronecker_delta_ik)
132  // sigma_kk = trace(sigma_ij)
133  // sigma_kk = 3 * sigma_0 where sigma_0 is the hydrostatic stress, pressure, or mean stress
134  // produces a scalar quantity that is invariant with respect to coordinates
135  return sigma_ij.tensor(0,0) + sigma_ij.tensor(1,1) + sigma_ij.tensor(2,2);
