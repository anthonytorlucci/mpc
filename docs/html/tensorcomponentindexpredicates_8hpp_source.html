198  template <typename S>
199  inline bool TensorRank4ComponentIndexIsAlias(const mpc::core::TensorRank4ComponentIndex& indexa,
200  const mpc::core::TensorRank4ComponentIndex& indexb) {
201  static_assert(std::is_base_of<mpc::core::SymmetryGroupBase,S>::value, "S must be derived from mpc::core::SymmetryGroupBase.");
202  // determines if indexa is an alias of indexb
203 
204  // by convention, mpc chooses to label an two indices that are equivalent as NOT being aliases. This follows the convention of "HasAlias"; if a
205  // component index of a tensor is only represented once, it does not have an alias. For example
206  // TensorRank4ComponentIndex(0,0,0,0) for TriclinicSymmetryGroupType does not have an alias and therefore it cannot be an alias of itself.
207  if (indexa == indexb) { return false; }
208 
209  // if neither index has an alias, they cannot be aliases of each other.
210  const bool indexa_has_alias = mpc::core::TensorRank4ComponentIndexHasAlias<S>(indexa);
211  const bool indexb_has_alias = mpc::core::TensorRank4ComponentIndexHasAlias<S>(indexb);
212  if (!indexa_has_alias || !indexb_has_alias) { return false; }
213 
214 
215  mpc::core::TensorRank4ComponentIndex a_reduced = mpc::core::ReducedTensorRank4ComponentIndex<S>(indexa);
216  mpc::core::TensorRank4ComponentIndex b_reduced = mpc::core::ReducedTensorRank4ComponentIndex<S>(indexb);
217  // FOR DEBUG std::cout << " a_reduced : " << a_reduced << ", b_reduced : " << b_reduced << std::endl;
218  return (a_reduced == b_reduced) ? true : false;
219  }
