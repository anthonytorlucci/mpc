208  int GetVoigtMatrixIndex(int ibase0, int jbase0) {
209  // NOTE:
210  // TODO: assert ibase1 and jbase1 belong to the set {1,2,3}
211  blitz::Array<int,2> kdelta = mpc::utilities::KroneckerDelta<int,3>();
212  int ibase1 = ibase0 + 1;
213  int jbase1 = jbase0 + 1;
214  int p = (ibase1 * kdelta(ibase0,jbase0)) + (9-ibase1-jbase1) * (1 - kdelta(ibase0,jbase0));
215  return p-1;
