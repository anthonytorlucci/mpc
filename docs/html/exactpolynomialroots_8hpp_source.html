26  std::array<std::complex<T>,2> QuadraticSolver(T a2, T a1, T a0) {
27  // solves the quadratic equation (a2)x**2 + (a1)x + (a0) = 0
28  //static_assert(std::is_arithmetic<T>::value, "Type T must be arithmetic");
29  static_assert(std::is_floating_point<T>::value, "Type T must be of type float, double, or long double");
30  std::complex<T> root0;
31  std::complex<T> root1;
32  T recip2a2 = 1.0 / (2.0 * a2); // leading coefficient
33  T sqrt_arg = (a1*a1) - (4.0 * a2 * a0); // the argument under the radical
34  if (sqrt_arg == 0) {
35  // both roots are real and equivalent
36  //std::cout << "both roots are real and equivalent" << std::endl;
37  root0.real(recip2a2 * (-1.0 * a1));
38  root0.imag(0.0);
39  root1.real(recip2a2 * (-1.0 * a1));
40  root1.imag(0.0);
41  }
42  if (sqrt_arg > 0) {
43  // both roots are real and distinct
44  //std::cout << "both roots are real and distinct" << std::endl;
45  root0.real(recip2a2 * ((-1.0 * a1) + sqrt(sqrt_arg)));
46  root0.imag(0.0);
47  root1.real(recip2a2 * ((-1.0 * a1) - sqrt(sqrt_arg)));
48  root1.imag(0.0);
49  }
50  if (sqrt_arg < 0) {
51  // both roots are imaginary
52  //std::cout << "both roots are imaginary" << std::endl;
53  root0.real(recip2a2 * (-1.0 * a1));
54  root0.imag(recip2a2 * sqrt(-1.0 * sqrt_arg));
55  root1.real(recip2a2 * (-1.0 * a1));
56  root1.imag(-1.0 * recip2a2 * sqrt(-1.0 * sqrt_arg)); // the -1 makes the argument positive; otherwise the sqrt(sqrt_arg) would return a nan.
57  }
58  return std::array<std::complex<T>,2>{root0, root1};
59  }
